from gtts import gTTS
import os
    
text = "hello my name is kamal and i am from uttarakhand "
language = 'en'
output = gTTS(text=text, lang=language, slow=False)
    
output.save("output.mp3")
os.system("mpg321 output.mp3") #for playing the audio
